from typing import List

def findRestaurant(list1: List[str], list2: List[str]) -> List[str]:
    
    dic = {}
    #result = []

    for i in range(len(list1)):
        for j in range(len(list2)):
            if list1[i] == list2[j]:
                dic[list1[i]] = i + j
    
    # for m, n in dic.items():

    #             result.append(m)
    # return result
    return [m for m, n in dic.items() if n == min(dic.values())]
