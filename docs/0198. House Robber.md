from typing import List
# Dynamic Programming
def rob(nums: List[int]) -> int:

    if len(nums) == 1:
        return nums[0]
    
    length = len(nums)
    dp = [0] * length

    dp[0] = nums[0]
    dp[1] = nums[1]

    for i in range(2, length):
        dp[i] = max(dp[i-1], dp[i-2] + nums[i])

    return dp[-1]

# recursive

    exclude = rob(nums[:-1])
    
    return max(include, exclude)
