from typing import List

# Union-Find
def union(n: int, connections: List[List[int]]) -> int:
    
    # the case there aren't enough cables
    if n - 1 > len(connections):
        return -1
    
    parent = list(range(n)) # [0,1,2,3]
    count = 0
        
    for x, y in connections:
        rootx = find(x, parent)
        rooty = find(y, parent)
        if rootx != rooty: # checking if 2 PC's are connecting
            parent[rootx] = rooty
            count += 1
        print(count,parent) # for understanding
 
    return n - 1 - count

def find(x, parent):

    if x != parent[x]:
        parent[x] = find(parent[x], parent)
    #print(parent, x) # for understanding
    return parent[x]
