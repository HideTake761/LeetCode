def nthUglyNumber(n: int) -> int:
    
    i, j = 1, 1 # i:pointer, j:counting ugly number

    while j < n:
        i += 1
        if i % 2 == 0 or i % 3 == 0 or i % 5 == 0:
            j += 1
    
    return i
