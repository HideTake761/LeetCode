from typing import Optional

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def reverseBetween(head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:

    if not head or left == right:
        return head

    stack = []

    while head:
        stack.append(head.val)
        head = head.next
    #print(stack) # for understanding

    list_reversed = list(reversed(stack[left-1:right]))
    #print(list_reversed) # for understanding

    stack = stack[:left-1] + list_reversed + stack[right:]
    #print(stack) # for understanding

    curr = ListNode(stack[0])
    head = curr

    for i in stack[1:]:
        temp = ListNode(i)
        curr.next = temp
        curr = temp
  
    while head:
        print(head.val, end = ' ') 
        head = head.next
    
    return head
