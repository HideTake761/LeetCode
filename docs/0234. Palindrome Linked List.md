class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def isPalindrome(keys) -> bool:

    temp = None

    for i in keys:
        head_r = ListNode(i)
        head_r.next = temp
        temp = head_r
        #print(head_r.val, 'reverse') # for understanding
        
    for j in keys:
        head = ListNode(j)
        #print(head_r.val) # for understanding
        if head.val != head_r.val:
            return False
        head_r = head_r.next
        
    return True
