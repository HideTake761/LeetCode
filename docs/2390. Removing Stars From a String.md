def removeStars(s: str) -> str:

    t = list(s)
    num = t.count('*')

    while num:
        idx = t.index('*')
        t.pop(idx)
        # Although it differs from the problem's constraints,
        # with this it still works even if the string starts with '*'.
        if idx > 0: t.pop(idx-1)
        num -= 1
    
    return "".join(t)
