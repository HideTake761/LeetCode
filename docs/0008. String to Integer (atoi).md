def myAtoi(s: str) -> int:

    sign, res = +1, ''

    for ch in s:
        if ch.isdigit():
            res += ch
        if ch == '-':
            sign = -1

    res = sign * int(res)
    
    return max(-2**31, min(res, 2**31 - 1))

if __name__ == '__main__':
