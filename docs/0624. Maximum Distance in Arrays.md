from typing import List

def maxDistance(arrays: List[List[int]]) -> int:

    n = len(arrays)
    res = 0
    # Each array is sorted in ascending order.
    # So we only use arrays[x][0] and arrays[y][-1] to find the maximum distance.
    for i in range(n):
        for j in range(n):
            if i != j: # two integers must be from two different arrays
                dis = abs(arrays[i][0] - arrays[j][-1])
                if dis > res: res = dis

    return res
