def sumRange(nums, left: int, right: int):

    return sum(nums[left:right+1])

def sumRange2(nums, left: int, right: int) -> int:
    
    prefix_sum = [0] * (len(nums))

    prefix_sum[0] = nums[0]

    for i in range(1, len(nums)):
        prefix_sum[i] = prefix_sum[i - 1] + nums[i]
    
    #print(prefix_sum) # for understanding

    return prefix_sum[right] - prefix_sum[left] + nums[left]
    # Using prefix_sum[right] - prefix_sum[left - 1] doesn't work when left = 0,
    # because it accesses index [-1].
