from typing import List

def nextGreaterElement(nums1: List[int], nums2: List[int]) -> List[int]:
    
    idx = []
    que = []
    
    for i in nums1:
        idx.append(nums2.index(i))
        #print(idx) # for understanding
    
    for j in idx:
        for k in range(j + 1, len(nums2)):
            if nums2[j] < nums2[k]:
                que.append(nums2[k])
                break
        else:
            que.append(-1)
    
    return que

            c = stack.pop()

        # {c(each element in num2): e(the next greater element to the right of c in num2)}
        stack.append(e)
        #print(stack, d) # for understanding
        
    return [d.get(e, -1) for e in nums1]
