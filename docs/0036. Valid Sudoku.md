from typing import List

# for checking repetition
def isValidSudoku(board: List[List[str]]) -> bool:

    res = []
    for i in range(9):
        for j in range(9):
            element = board[i][j]
            if element != '.':
                # row, column, 3x3 sub-boxes
                res += [(i, element), (element, j), (i // 3, j // 3, element)]
    #print(res) # for understanding
    #print(set(res)) # for understanding
    
    return len(res) == len(set(res))

# for checking "must contain the digits 1-9"
def isValidSudoku2(board: List[List[str]]) -> bool:

    # checking row
    for i in range(9):
        if board[i].count('.') == 9:
            return False
    
    # transpose the matrix board
    board_transposed = [list(x) for x in zip(*board)]
    #print(board_transposed) # for understanding

    # checking column
    for j in range(9):
        if board_transposed[j].count('.') == 9:
            return False

    # checking 3x3 sub-boxes
    for m in range(3):
        for n in range(3):
            tmp = []
            for x in range(m*3, (m+1)*3):            
                for y in range(n*3, (n+1)*3):
                    tmp.append(board[x][y])
                    #print(tmp) # for understanding
                    if tmp.count('.') == 9:
                        return False
 
    return True
