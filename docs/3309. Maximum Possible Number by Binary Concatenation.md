from typing import List

def maxGoodNumber(nums: List[int]) -> int:

    binary_list = []

    for i in nums:
        binary_list.append(format(i, 'b'))
    
    for n in range(len(binary_list)-1, 0, -1):
        i = 0
        while i < n:
            if int(binary_list[i]+binary_list[i+1]) < \
                int(binary_list[i+1]+binary_list[i]):
                binary_list[i], binary_list[i+1] = binary_list[i+1], binary_list[i]
            i += 1
            #print(i, n, binary_list) # for understanding

    return int("".join([i for i in binary_list]), 2)

if __name__ == "__main__":
